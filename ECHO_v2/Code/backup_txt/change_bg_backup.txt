from tkinter import *
from PIL import ImageTk,Image
from tkinter import filedialog

def file():
    global filename, img1
    filename = filedialog.askopenfilename(initialdir="/files", title="select a file",
                                                filetypes=(("png files", "*.png"), ("jpeg files", "*.jpeg")))
    print("you have selected" + filename)
    
    img1 = ImageTk.PhotoImage(Image.open(filename))
    redbus = img1
    l2.config(text= filename)
    l1.config(image= redbus)
    return redbus
    

#file() # calling from main / button 

def main():
    root = Tk()
    root.geometry("800x500")
    global redbus

    ok = Image.open("C:/Users/mcanu/Desktop/Codemy/Images/a1.png")
    redbus = ImageTk.PhotoImage(ok)

    global l1, l2
    l1 = Label(root, image= redbus, height=500, width=800)
    l1.place(x=0, y=0)

    Button(l1, text= "select a file", command= file).place(x=30, y=30)

    l2 = Label(l1, text="")
    l2.place(x=100, y=100)


    root.mainloop()



main()
