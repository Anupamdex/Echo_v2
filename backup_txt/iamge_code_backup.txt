
from tkinter import *
import time
from PIL import ImageTk,Image


def app(w_value, h_value):

    win = Tk()
    win.geometry(w_value+"x"+h_value)
    
    print("import app here")
    C1 = Canvas(win, background="red", height= h_value, width= w_value, highlightthickness= 0)
    C1.place(x=0, y=0)
    Label(C1, text="hey").place(x=0, y=0)

    movie = Image.open("files/background.png")

    #OPEN and RESIZE image
    resized = movie.resize(((int(w_value)), (int(h_value))), Image.Resampling.LANCZOS)

    sam = ImageTk.PhotoImage(resized)

    C1.create_image(0,0, anchor= NW, image= sam)

    win.mainloop()
